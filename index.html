                        static void Header(string title)
                        {
                            Console.WriteLine();
                            Console.ForegroundColor = ConsoleColor.Yellow;

                        // POPRAWKA #2 (zastosowana we wszystkich metodach):
                        // wait.Until zwraca element gdy pojawi sie w DOM,
                        // ale element moze byc ukryty lub nieaktywny.
                        // Sprawdzamy Displayed && Enabled przed interakcja.

                        static void Fill(IWebDriver driver, WebDriverWait wait,
                                        string fieldId, string value, string label)
                        {
                            try
                            {
                                var el = wait.Until(d => {
                                    var e = d.FindElement(By.Id(fieldId));
                                    return (e.Displayed && e.Enabled) ? e : null;
                                });
                                el.Clear();
                                el.SendKeys(value);
                                Ok($"{label}: {value}");
                            }
                            catch
                            {
                                Fail($"{label}: pole '{fieldId}' nie znalezione! " +
                                    $"By.Id(\"{fieldId}\") => NoSuchElementException");
                            }
                        }

                        static void FillDate(IWebDriver driver, WebDriverWait wait,
                                            string fieldId, string isoDate, string label)
                        {
                            // input[type="date"] wymaga wartości w formacie yyyy-MM-dd.
                            // Najniezawodniej ustawiamy przez JavaScript, bo locale przeglądarki
                            // może powodować problemy z SendKeys na różnych systemach.
                            try
                            {
                                wait.Until(d => {
                                    var e = d.FindElement(By.Id(fieldId));
                                    return (e.Displayed && e.Enabled) ? e : null;
                                });
                                ((IJavaScriptExecutor)driver).ExecuteScript(
                                    $"var el = document.getElementById('{fieldId}');" +
                                    $"el.value = '{isoDate}';" +
                                    $"el.dispatchEvent(new Event('change'));");
                                Ok($"{label}: {isoDate}");
                            }
                            catch
                            {
                                Fail($"{label}: pole '{fieldId}' nie znalezione!");
                            }
                        }

                        static void SelectDropdown(IWebDriver driver, WebDriverWait wait,
                                                    string fieldId, string visibleText, string label)
                        {
                            try
                            {
                                var el = wait.Until(d => {
                                    var e = d.FindElement(By.Id(fieldId));
                                    return (e.Displayed && e.Enabled) ? e : null;
                                });
                                var select = new SelectElement(el);
                                select.SelectByText(visibleText);
                                Ok($"{label}: {visibleText}");
                            }
                            catch
                            {
                                Fail($"{label}: dropdown '{fieldId}' lub opcja '{visibleText}' nie znaleziona!");
                            }
                        }

                        static void ClickRadio(IWebDriver driver, WebDriverWait wait,
                                            string radioName, string value, string label)
                        {
                            // Radiobutton ma display:none — klikamy rodzica .radio-item przez JS
                            try
                            {
                                var parent = wait.Until(d =>
                                    d.FindElement(By.XPath(
                                        $"//input[@name='{radioName}' and @value='{value}']/..")));
                                ((IJavaScriptExecutor)driver).ExecuteScript("arguments[0].click();", parent);
                                Ok($"{label}: {value}");
                            }
                            catch
                            {
                                Fail($"{label}: radio name='{radioName}' value='{value}' nie znaleziony!");
                            }
                        }

                        static void ClickCheckbox(IWebDriver driver, WebDriverWait wait,
                                                string checkboxName, string value, string label)
                        {
                            // Checkbox ma display:none — klikamy rodzica .check-item przez JS.
                            // Klikamy tylko jeśli element nie jest jeszcze oznaczony jako checked.
                            try
                            {
                                var parent = wait.Until(d =>
                                    d.FindElement(By.XPath(
                                        $"//input[@name='{checkboxName}' and @value='{value}']/..")));

                                var cssClass = parent.GetAttribute("class") ?? "";
                                if (!cssClass.Contains("checked"))
                                    ((IJavaScriptExecutor)driver).ExecuteScript("arguments[0].click();", parent);

                                Ok($"{label}");
                            }
                            catch
                            {
                                Fail($"{label}: checkbox name='{checkboxName}' value='{value}' nie znaleziony!");
                            }
                        }

                        static void ClickButton(IWebDriver driver, WebDriverWait wait,
                                                string xpath, string label)
                        {
                            try
                            {
                                Console.WriteLine();
                                var btn = wait.Until(d => {
                                    var e = d.FindElement(By.XPath(xpath));
                                    return (e.Displayed && e.Enabled) ? e : null;
                                });
                                btn.Click();
                                Ok($"Kliknieto: {label}");
                            }
                            catch
                            {
                                Fail($"Przycisk nie znaleziony: {label}  xpath={xpath}");
                            }
                        }

                        static void ReadReview(IWebDriver driver, string elementId, string label)
                        {
                            try
                            {
                                var text = driver.FindElement(By.Id(elementId)).Text;
                                Console.ForegroundColor = ConsoleColor.DarkGray;
                                Console.WriteLine($"  ℹ  {label}: {text}");
                                Console.ResetColor();
                            }
                            catch
                            {
                                Console.ForegroundColor = ConsoleColor.DarkGray;
                                Console.WriteLine($"  ℹ  {label}: (brak elementu #{elementId})");
                                Console.ResetColor();
                            }
                        }

                        static void Ok(string msg)
                        {
                            Console.ForegroundColor = ConsoleColor.Green;
                            Console.Write("  OK ");
                            Console.ResetColor();
                            Console.WriteLine(msg);
                        }

                        static void Fail(string msg)
                        {
                            Console.ForegroundColor = ConsoleColor.Red;
                            Console.Write("  XX ");
                            Console.ResetColor();
                            Console.WriteLine(msg);
                        }

                        public static class PlaywrightProcessResolver

                            public static (string Command, string[] Arguments) Resolve(bool headless)
                            {

                                {
                                    return ("npx", BuildArgs(headless));
                                }

                                var candidates = new[]
                                {
                                    Path.Combine(
                                        Environment.GetFolderPath

                                    Path.Combine(
                                        Environment.GetFolderPath
                                            (Environment.SpecialFolder.ProgramFilesX86), "nodejs", "npx.cmd"),
                                    Path.Combine(
                                        Environment.GetFolderPath
                                            (Environment.SpecialFolder.ApplicationData), "nvm", "current", "npx.cmd"),
                                    Path.Combine(
                                        Environment.GetFolderPath
                                            (Environment.SpecialFolder.ApplicationData), "nvm", "v24.14.0", "npx.cmd"),
                                    Path.Combine(
                                        Environment.GetFolderPath
                                            (Environment.SpecialFolder.ApplicationData), "nvm", "v22.0.0", "npx.cmd"),
                                };

                                var npxCmd = candidates.FirstOrDefault(File.Exists);

                                if (npxCmd != null)
                                {
                                    return (npxCmd, BuildArgs(headless));
                                }

                                var fallbackArgs = new List<string> { "/c", "npx" };
                                fallbackArgs.AddRange(BuildArgs(headless));
                                return ("cmd", fallbackArgs.ToArray());
                            }

                            public static async Task<bool> VerifyAsync(string command, string[] arguments)
                            {
                                try
                                {
                                    var proc = new Process
                                    {
                                        StartInfo = new ProcessStartInfo
                                        {
                                            FileName = command,
                                            Arguments = string.Join(" ", arguments.Select(a => a.Contains(' ') ? $"\"{a}\"" : a)),
                                            RedirectStandardError = true,
                                            UseShellExecute = false,
                                            CreateNoWindow = true,
                                        }
                                    };

                                    proc.Start();
                                    await Task.Delay(500);

                                    if (proc.HasExited)
                                    {
                                        var stderr = await proc.StandardError.ReadToEndAsync();
                                        ConsoleUi.WriteError("Proces @playwright/mcp zakonczyl sie natychmiast!");
                                        if (proc.ExitCode != 0)
                                            ConsoleUi.WriteError($"Exit code: {proc.ExitCode}");
                                        if (!string.IsNullOrWhiteSpace(stderr))
                                            ConsoleUi.WriteError($"Stderr: {stderr}");

                                        Console.WriteLine();
                                        Console.WriteLine("Najczestsza przyczyna: brak przegladarek Playwright");
                                        Console.WriteLine("Fix: npx playwright install chromium");
                                        Console.WriteLine();
                                        return false;
                                    }

                                    proc.Kill(entireProcessTree: true);
                                    return true;
                                }
                                catch (Exception ex)
                                {
                                    ConsoleUi.WriteError($"Nie mozna uruchomic npx: {ex.Message}");
                                    return false;
                                }
                            }

                            private static string[] BuildArgs(bool headless)
                            {
                                return headless
                                    ? ["-y", "@playwright/mcp@latest", "--headless"]
                                    : ["-y", "@playwright/mcp@latest"];

                        ## 🚨 PROBLEMY KRYTYCZNE (WCAG A)

                        ### 1. **Checkboxy bez aria-label / labelledby - WSZYSTKIE KROKI**
                        **Lokalizacja:** 
                        - Krok 1: Checkbox "Send me occasional emails..."
                        - Krok 2: Wszystkie radio buttons (Contract Type, Remote/On-site)
                        - Krok 3: 8 checkboxów (Email Backup, 2FA, VPN, etc.)
                        - Krok 4: Checkbox "Confirm the conditions"

                        **WCAG:** 1.3.1 Info and Relationships (Level A)

                        **Problem:** Żaden z checkboxów/radio buttons nie ma:
                        - `id` atrybutu
                        - `aria-label` atrybutu
                        - `aria-labelledby` atrybutu
                        - Powiązanego `<label for="id">` elementu

                        **Impact:** Ekrany dla osób niewidomych (screen readers) nie mogą przeczytać nazwy kontrolki. Użytkownik nie wie co zaznacza.

                        **Rekomendacja:**
                        ```html
                        <!-- ŹLE - aktualnie -->
                        <input type="checkbox" name="email_optins">
                        Send me occasional emails...

                        <!-- DOBRZE - powinno być -->
                        <div>
                        <input type="checkbox" id="email_optins" name="email_optins" aria-label="Send me occasional emails about company updates">
                        <label for="email_optins">Send me occasional emails about company updates</label>
                        </div>

                        <!-- LUB alternatywa -->
                        <input type="checkbox" name="email_optins" aria-label="Send me occasional emails about company updates">
                        ```

                        **Priorytet:** 🔴 NATYCHMIAST - blokuje wszystkie użytkowników screen readers

                        ### 2. **Brak `<form>` elementu - CAŁA STRONA**
                        **Lokalizacja:** Główny kontener formularza

                        **WCAG:** 4.1.2 Name, Role, Value (Level A)

                        **Problem:** Cały formularz to `<div>` z `id="multistep-form"`. Brakuje semantycznego elementu `<form>`.

                        **Impact:** 
                        - Assistive technologies nie rozpoznają to jako formularz
                        - Brak wbudowanego submit behavior
                        - Screen readers nie informują o tym że to form

                        <!-- ŹLE -->
                        <div id="multistep-form" class="form-container">

                        </div>

                        <form id="multistep-form" class="form-container" aria-label="Employee Onboarding Form">

                        # Playwright Agent Instructions

                        You operate in one of two modes depending on the user's request.
                        Read the user's message carefully and select the correct mode.

                        ---

                        ## MODE 1 — Test Generator

                        **Trigger:** User provides a scenario or feature to test (e.g. "generate a test for the onboarding form").

                        ### Workflow
                        1. DO NOT generate test code based on the scenario description alone
                        2. Navigate to the page using Playwright MCP tools
                        3. Run each step one by one using the tools provided
                        4. Only after all steps are completed, emit a Playwright TypeScript test based on the message history
                        5. Save the generated test file in the `tests/` directory
                        6. Execute the test file and iterate until the test passes

                        ### Rules for generated tests
                        - Use `getByLabel()`, `getByRole()`, `getByText()` — NEVER CSS selectors or XPath
                        - Use meaningful names for test files and test cases
                        - Use `async/await` syntax for all asynchronous operations
                        - Use Playwright's built-in assertions (`expect`) to verify expected outcomes
                        - Group related scenarios in `describe()` blocks

                        ---

                        ## MODE 2 — Test Healer

                        **Trigger:** User asks to fix a failing test, or provides an error message from a Playwright run.

                        ### Workflow
                        1. Read the current test file (`tests/generated.spec.ts` unless told otherwise)
                        2. Run the test using the Playwright MCP tool
                        3. If the test **PASSES** — report success and stop
                        4. If the test **FAILS**:
                        - Analyse the exact error message
                        - If needed, navigate the page with Playwright MCP to inspect the DOM
                        - Edit the test file directly — make the **minimal fix** required
                        - Run the test again
                        - Repeat until the test passes or 3 attempts are exhausted

                        ### Rules for fixes
                        - Use `getByLabel()`, `getByRole()`, `getByText()` — NEVER CSS selectors or XPath
                        - If `getByText()` causes a **strict mode violation** (resolves to multiple elements),
                        use `page.locator('#specific-id')` or `.first()` to narrow the match
                        - Do NOT rewrite the whole test — minimal changes only
                        - Do NOT explain what you are doing — just act
